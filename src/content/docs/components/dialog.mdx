## Usage

```tsx
import {
 DialogContent,
 DialogDescription,
 DialogFooter,
 DialogHeader,
 DialogOverlay,
 DialogTitle,
 DialogTrigger,
} from "@/registry/ui/dialog";
```

```tsx
<DialogTrigger>
    <Button />
    <DialogOverlay>
        <DialogContent>
            <DialogHeader>
                <DialogTitle />
                <DialogDescription />
            </DialogHeader>
                ...content
            <DialogFooter>
                <Button type="submit" />
            </DialogFooter>
        </DialogContent>
    </DialogOverlay>
</DialogTrigger>
```
