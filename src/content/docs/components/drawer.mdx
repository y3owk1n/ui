## Usage

```tsx
import { Button } from "@/registry/ui/button";
import {
 DrawerContent,
 DrawerDescription,
 DrawerFooter,
 DrawerHeader,
 DrawerOverlay,
 DrawerTitle,
 DrawerTrigger,
} from "@/registry/ui/drawer";
```

```tsx
<DrawerTrigger>
    <Button />
    <DrawerOverlay>
        <DrawerContent>
            <div className="mx-auto max-w-[425px]">
                <DrawerHeader>
                    <DrawerTitle />
                    <DrawerDescription />
                </DrawerHeader>
                    ...content
                <DrawerFooter />
                    <Button type="submit" />
                </DrawerFooter>
            </div>
        </DrawerContent>
    </DrawerOverlay>
</DrawerTrigger>
```
