## Usage

```tsx
import { CheckboxGroup } from "@/components/ui/checkbox-group"
import { Label } from "@/components/ui/label"
import { FieldError } from "@/components/ui/field-error"
```

```tsx
<CheckboxGroup>
    <Label>Label</Label>
    <Checkbox value="value">Value</Checkbox>
    <FieldError />
</CheckboxGroup>
```
