## Usage

```tsx
import { DateField, DateInput, DateSegment } from "@/components/ui/date-field";
```

```tsx
<DateField>
    <Label />
    <DateInput>
        {(segment) => <DateSegment segment={segment} />}
    </DateInput>
    <FieldDescription>Date of your choice</FieldDescription>
    <FieldError />
</DateField>
```
