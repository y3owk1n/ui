## Usage

```tsx
import {
  ComboBox,
  ComboBoxContent,
  ComboBoxItem,
  ComboBoxLabel,
  ComboBoxPopover,
  ComboBoxTrigger,
} from "@/components/ui/combo-box";
```

```tsx
<ComboBox defaultItems={rows}>
  <ComboBoxLabel />
  <ComboBoxTrigger placeholder="Select something" />
  <ComboBoxPopover>
    <ComboBoxContent<(typeof rows)[number]>>
      {(item) => <ComboBoxItem textValue={item.name}>{item.name}</ComboBoxItem>}
    </ComboBoxContent>
  </ComboBoxPopover>
</ComboBox>
```
